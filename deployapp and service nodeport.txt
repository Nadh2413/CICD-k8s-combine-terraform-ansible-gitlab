apiVersion: apps/v1
kind: Deployment
metadata:
  name: cafe-app-deployment
  labels:
    app: cafe-app
spec:
  replicas: 2
  selector:
    matchLabels:
      app: cafe-app
  template:
    metadata:
      labels:
        app: cafe-app
    spec:
      containers:
      - name: cafe-app
        image: cdhuan1003/cafe-app
        ports:
        - containerPort: 80
